Run on Blackview (or any host with kubectl + your Vault root token). Replace ROOT_TOKEN.

export KUBECONFIG=$HOME/.kube/config
export VAULT_TOKEN=ROOT_TOKEN

# 1) Enable kubernetes auth (idempotent)
kubectl exec -n vault vault-0 -c vault -- env VAULT_TOKEN="$VAULT_TOKEN" vault auth enable kubernetes 2>/dev/null || true

# 2) Configure auth (uses the vault pod service account as token reviewer; requires Vault SA to have TokenReview perms — default Helm chart usually grants this)
kubectl exec -n vault vault-0 -c vault -- env VAULT_TOKEN="$VAULT_TOKEN" vault write auth/kubernetes/config \
  kubernetes_host="https://kubernetes.default.svc:443" \
  kubernetes_ca_cert=@/var/run/secrets/kubernetes.io/serviceaccount/ca.crt \
  token_reviewer_jwt=@/var/run/secrets/kubernetes.io/serviceaccount/token

# 3) Policy: KV v2 paths for threat-intel
kubectl exec -n vault vault-0 -c vault -- env VAULT_TOKEN="$VAULT_TOKEN" vault policy write vso-policy - <<'EOF'
path "secret/data/threat-intel/*" {
  capabilities = ["read", "list"]
}
path "secret/metadata/threat-intel/*" {
  capabilities = ["read", "list"]
}
EOF

# 4) Role for VSO service account in namespace vault-sep
kubectl exec -n vault vault-0 -c vault -- env VAULT_TOKEN="$VAULT_TOKEN" vault write auth/kubernetes/role/vso-role \
  bound_service_account_names=vault-secrets-operator \
  bound_service_account_namespaces=vault-sep \
  policies=vso-policy \
  ttl=24h

unset VAULT_TOKEN
